The libfreerdp Core API is the central entry point for building RDP client applications. It provides the `freerdp` instance abstraction that manages a single RDP connection end-to-end — from initial setup through connect, event processing, and disconnect — along with a rich set of callback hooks for authentication, certificate verification, and channel data transfer.

## Linking Against libfreerdp3

FreeRDP 3.x installs pkg-config and CMake package files so you can integrate it into any build system without hard-coding paths.

<Tabs>
  <Tab title="pkg-config / Makefile">
    ```bash
    # Compile and link a single-file client
    cc my_client.c \
        $(pkg-config --cflags --libs freerdp3 freerdp-client3 winpr3) \
        -o my_client
    ```
  </Tab>
  <Tab title="CMake find_package">
    ```cmake
    cmake_minimum_required(VERSION 3.14)
    project(my_rdp_client C)

    find_package(FreeRDP3 REQUIRED)
    find_package(FreeRDP-Client3 REQUIRED)
    find_package(WinPR3 REQUIRED)

    add_executable(my_client my_client.c)
    target_link_libraries(my_client
        freerdp3
        freerdp-client3
        winpr3
    )
    ```
  </Tab>
</Tabs>

## Instance Lifecycle

A FreeRDP client connection follows a strict lifecycle. Every allocation must be matched by its corresponding free function, and context/callbacks must be configured before `freerdp_connect` is called.

```
freerdp_new()
    └─► set ContextSize / callbacks (ContextNew, ContextFree)
freerdp_context_new()
    └─► configure rdpSettings via freerdp_settings_set_*
freerdp_connect()
    └─► event loop: freerdp_check_event_handles()
freerdp_disconnect()
freerdp_context_free()
freerdp_free()
```

### Allocation and Deallocation

```c
#include <freerdp/freerdp.h>
#include <freerdp/client.h>

/* Allocate a new freerdp instance — sets default callbacks */
freerdp* instance = freerdp_new();

/* Optionally extend the context with application-specific data */
instance->ContextSize = sizeof(MyClientContext); /* must be >= sizeof(rdpContext) */
instance->ContextNew  = my_context_new_cb;
instance->ContextFree = my_context_free_cb;

/* Allocate the context (and the embedded rdpSettings) */
if (!freerdp_context_new(instance))
    goto fail;

/* ... configure settings, set callbacks ... */

/* Connect */
if (!freerdp_connect(instance))
    goto fail;

/* Event loop */
while (!freerdp_shall_disconnect_context(instance->context))
{
    DWORD count;
    HANDLE handles[MAXIMUM_WAIT_OBJECTS] = { 0 };
    count = freerdp_get_event_handles(instance->context, handles, ARRAYSIZE(handles));

    WaitForMultipleObjects(count, handles, FALSE, INFINITE);
    if (!freerdp_check_event_handles(instance->context))
        break;
}

freerdp_disconnect(instance);

fail:
    freerdp_context_free(instance);
    freerdp_free(instance);
```

### Core Lifecycle Functions

| Function | Signature | Description |
|---|---|---|
| `freerdp_new` | `freerdp* freerdp_new(void)` | Allocate a new instance |
| `freerdp_free` | `void freerdp_free(freerdp*)` | Deallocate the instance |
| `freerdp_context_new` | `BOOL freerdp_context_new(freerdp*)` | Allocate context and settings |
| `freerdp_context_new_ex` | `BOOL freerdp_context_new_ex(freerdp*, rdpSettings*)` | Allocate context with pre-built settings |
| `freerdp_context_free` | `void freerdp_context_free(freerdp*)` | Free the context |
| `freerdp_connect` | `BOOL freerdp_connect(freerdp*)` | Begin the RDP connection |
| `freerdp_disconnect` | `BOOL freerdp_disconnect(freerdp*)` | Cleanly disconnect |
| `freerdp_reconnect` | `BOOL freerdp_reconnect(freerdp*)` | Reconnect using current settings |
| `freerdp_abort_connect_context` | `BOOL freerdp_abort_connect_context(rdpContext*)` | Abort a connection in progress |

## The rdp_context Structure

`rdpContext` is embedded in every `freerdp` instance via `instance->context`. It carries all live connection state and is the primary object passed to most API functions.

```c
struct rdp_context
{
    ALIGN64 freerdp*      instance;    /* back-pointer to the owning instance */
    ALIGN64 freerdp_peer* peer;        /* server-side peer (server mode only) */
    ALIGN64 BOOL          ServerMode;  /* TRUE when operating as a server */
    ALIGN64 UINT32        LastError;

    ALIGN64 int    argc;               /* saved command-line argument count */
    ALIGN64 char** argv;               /* saved command-line arguments */

    ALIGN64 rdpRdp*        rdp;        /* internal RDP engine */
    ALIGN64 rdpGdi*        gdi;        /* GDI rendering surface */
    ALIGN64 rdpSettings*   settings;   /* connection settings (owned by rdpRdp) */
    ALIGN64 rdpChannels*   channels;   /* virtual channel manager */
    ALIGN64 rdpInput*      input;      /* input injection interface */
    ALIGN64 rdpUpdate*     update;     /* display update callbacks */
    ALIGN64 rdpCodecs*     codecs;     /* codec negotiation state */
    ALIGN64 rdpAutoDetect* autodetect; /* bandwidth/RTT auto-detection */
    ALIGN64 wLog*          log;        /* per-connection WLog logger */
};
```

Extend the context for application state by setting `instance->ContextSize` before calling `freerdp_context_new`:

```c
typedef struct {
    rdpContext context;   /* MUST be first */
    HWND       window;
    int        myData;
} MyClientContext;

instance->ContextSize = sizeof(MyClientContext);
```

## The rdpClientContext Structure

For full client-side use, the higher-level `rdpClientContext` embeds `rdpContext` and adds common client helpers from `<freerdp/client.h>`:

```c
struct rdp_client_context
{
    rdpContext  context;          /* embedded base context — MUST be first */
    ALIGN64 HANDLE thread;        /* client worker thread handle */
    ALIGN64 INT32  lastX;         /* last known pointer X position */
    ALIGN64 INT32  lastY;         /* last known pointer Y position */
    ALIGN64 BOOL   mouse_grabbed; /* whether the pointer is currently captured */

    ALIGN64 FreeRDP_TouchContact contacts[FREERDP_MAX_TOUCH_CONTACTS];
    ALIGN64 FreeRDP_PenDevice    pens[FREERDP_MAX_PEN_DEVICES];
};
```

Use `freerdp_client_context_new` / `freerdp_client_context_free` when building on the client-common library.

## Connection Callbacks

Set callbacks on the `freerdp` instance **before** calling `freerdp_connect`. All function-pointer fields are declared with `WINPR_ATTR_NODISCARD`; always check their return values.

### Lifecycle Callbacks

```c
/* Called after freerdp_context_new allocates and zeroes the context buffer */
typedef BOOL (*pContextNew)(freerdp* instance, rdpContext* context);

/* Called before freerdp_context_free releases memory */
typedef void (*pContextFree)(freerdp* instance, rdpContext* context);

/* Called before the TCP/TLS handshake; used to apply final settings */
typedef BOOL (*pConnectCallback)(freerdp* instance);  /* also PostConnect, Redirect */

/* Called after disconnect, before channels are torn down */
typedef void (*pPostDisconnect)(freerdp* instance);
```

### Authentication Callback

The preferred authentication callback since FreeRDP 3.x is `pAuthenticateEx`, which receives the **reason** for the prompt so a single handler can cover all credential types:

```c
typedef enum {
    AUTH_NLA,           /* NLA (Kerberos/NTLM) credentials */
    AUTH_TLS,           /* TLS-level authentication */
    AUTH_RDP,           /* Classic RDP security */
    GW_AUTH_HTTP,       /* HTTP gateway */
    GW_AUTH_RDG,        /* RD Gateway */
    GW_AUTH_RPC,        /* RPC gateway */
    AUTH_SMARTCARD_PIN, /* Smartcard PIN prompt */
    AUTH_RDSTLS,        /* RDSTLS authentication */
    AUTH_FIDO_PIN       /* FIDO2 authenticator PIN (since 3.x) */
} rdp_auth_reason;

typedef BOOL (*pAuthenticateEx)(freerdp* instance,
                                char**   username,
                                char**   password,
                                char**   domain,
                                rdp_auth_reason reason);
```

<Note>
The input `*username`, `*password`, and `*domain` strings are heap-allocated. Free the old value with `free()` before replacing it with a newly allocated string. Return `TRUE` to continue the connection (even with empty credentials) or `FALSE` to abort it.
</Note>

Assign it on the instance:

```c
static BOOL my_authenticate(freerdp* instance, char** user, char** pass,
                             char** domain, rdp_auth_reason reason)
{
    free(*user);
    *user = strdup("alice");
    free(*pass);
    *pass = strdup("s3cr3t");
    return TRUE;
}

instance->AuthenticateEx = my_authenticate;
```

## Certificate Verification Callbacks

### Certificate Verification Flags

These flags are passed to the `pVerifyCertificateEx` and `pVerifyChangedCertificateEx` callbacks:

```c
#define VERIFY_CERT_FLAG_NONE             0x00  /* normal, first-time certificate */
#define VERIFY_CERT_FLAG_LEGACY           0x02  /* legacy (pre-3.x) verification path */
#define VERIFY_CERT_FLAG_REDIRECT         0x10  /* certificate seen during redirection */
#define VERIFY_CERT_FLAG_GATEWAY          0x20  /* RD Gateway certificate */
#define VERIFY_CERT_FLAG_CHANGED          0x40  /* certificate has changed since last visit */
#define VERIFY_CERT_FLAG_MISMATCH         0x80  /* subject does not match hostname */
#define VERIFY_CERT_FLAG_MATCH_LEGACY_SHA1 0x100 /* fingerprint matches stored SHA-1 */
#define VERIFY_CERT_FLAG_FP_IS_PEM        0x200 /* fingerprint parameter is PEM data */
```

### pVerifyCertificateEx

Called when connecting to a host whose certificate is not yet trusted:

```c
typedef DWORD (*pVerifyCertificateEx)(
    freerdp*    instance,
    const char* host,
    UINT16      port,
    const char* common_name,
    const char* subject,
    const char* issuer,
    const char* fingerprint,   /* SHA-256 hex or PEM if VERIFY_CERT_FLAG_FP_IS_PEM */
    DWORD       flags);

/* Return values:
 *  0 — reject: abort connection
 *  1 — accept and store permanently
 *  2 — accept for this session only
 */
```

### pVerifyChangedCertificateEx

Called when the server presents a certificate different from the stored one (potential MITM):

```c
typedef DWORD (*pVerifyChangedCertificateEx)(
    freerdp*    instance,
    const char* host,       UINT16 port,
    const char* common_name,
    const char* subject,    const char* issuer,    const char* new_fingerprint,
    const char* old_subject, const char* old_issuer, const char* old_fingerprint,
    DWORD       flags);
```

### pVerifyX509Certificate

Provides the full PEM certificate chain for custom validation logic:

```c
typedef int (*pVerifyX509Certificate)(
    freerdp*    instance,
    const BYTE* data,       /* DER-encoded certificate chain */
    size_t      length,
    const char* hostname,
    UINT16      port,
    DWORD       flags);
```

## Minimal Client Example

The following snippet shows the minimal wiring needed to establish a connection:

```c
#include <freerdp/freerdp.h>
#include <freerdp/client.h>
#include <freerdp/settings.h>

static BOOL my_pre_connect(freerdp* instance)
{
    rdpSettings* s = instance->context->settings;
    freerdp_settings_set_string(s, FreeRDP_ServerHostname, "rdp.example.com");
    freerdp_settings_set_uint32(s, FreeRDP_ServerPort,     3389);
    freerdp_settings_set_string(s, FreeRDP_Username,       "alice");
    freerdp_settings_set_string(s, FreeRDP_Password,       "s3cr3t");
    freerdp_settings_set_uint32(s, FreeRDP_DesktopWidth,   1920);
    freerdp_settings_set_uint32(s, FreeRDP_DesktopHeight,  1080);
    return TRUE;
}

int main(void)
{
    freerdp* instance = freerdp_new();
    instance->PreConnect          = my_pre_connect;
    instance->VerifyCertificateEx = client_cli_verify_certificate_ex;
    instance->AuthenticateEx      = client_cli_authenticate_ex;

    if (!freerdp_context_new(instance))
        return 1;

    if (!freerdp_connect(instance))
        return 1;

    while (!freerdp_shall_disconnect_context(instance->context))
    {
        HANDLE handles[64];
        DWORD  count = freerdp_get_event_handles(instance->context,
                                                  handles, 64);
        WaitForMultipleObjects(count, handles, FALSE, 100);
        freerdp_check_event_handles(instance->context);
    }

    freerdp_disconnect(instance);
    freerdp_context_free(instance);
    freerdp_free(instance);
    return 0;
}
```
